123  void update(const Vector3D &accel, const Vector3D &gyro, const double time,
124  const double favoring) {
125  // math from:
126  // https://stanford.edu/class/ee267/lectures/lecture10.pdf
127  // https://stanford.edu/class/ee267/notes/ee267_notes_imu.pdf
128 
129  // rotation quaternion from gyroscope readings
130  Quaternion qGyroCur;
131  if (MathUtil::nearZero(gyro)) {
132  // if gyro reading is 0, take current quaternion to be corrected
133  qGyroCur = m_qRot;
134  } else {
135  // otherwise integrate quaternion reading
136  const Quaternion qGyroDelta{normalize(gyro), time * magn(gyro)};
137  qGyroCur = m_qRot * qGyroDelta;
138  }
139 
140  // don't bother with acceleration correction if acceleration is basically
141  // 0
142  if (MathUtil::nearZero(accel)) {
143  m_qRot = qGyroCur;
144  return;
145  }
146 
147  // gravity vector rotation
148  const Quaternion qAccelBody{0, accel};
149  const Quaternion qAccelWorld =
150  qGyroCur * qAccelBody *
151  qGyroCur.conj(); // rotates body acceleration by gyro measurements
152 
153  // correcting gyro drift with accelerometer
154  const Vector3D vecAccelWorldNorm = normalize(qAccelWorld.vec());
155  const Vector3D vecAccelGravity{0, 0, -1};
156  const Vector3D vecRotAxis =
157  cross(vecAccelWorldNorm,
158  vecAccelGravity); // rotation axis for correction rotation from
159  // estimated gravity vector (from gyro
160  // readings) to true gravity vector
161  const double rotAngle = std::acos(
162  MathUtil::clamp(dot(vecAccelGravity, vecAccelWorldNorm) /
163  (magn(vecAccelGravity) * magn(vecAccelWorldNorm)),
164  -1.0,
165  1.0)); // angle to rotate to correct acceleration vector
166 
167  // if angle needed to rotate is 0 or the axis to rotate around is 0, then
168  // don't bother correcting
169  if (MathUtil::nearZero(rotAngle) || MathUtil::nearZero(vecRotAxis)) {
170  m_qRot = qGyroCur;
171  return;
172  }
173 
174  // complementary filter
175  const Quaternion qAccelCur{normalize(vecRotAxis),
176  (1 - favoring) * rotAngle};
177  m_qRot = qAccelCur * qGyroCur;
178  }
