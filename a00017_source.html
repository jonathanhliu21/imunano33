147  void update(const Vector3D &accel, const Vector3D &gyro, const num_t time,
148  const num_t favoring) {
149  // math from:
150  // https://stanford.edu/class/ee267/lectures/lecture10.pdf
151  // https://stanford.edu/class/ee267/notes/ee267_notes_imu.pdf
152 
153  // rotation quaternion from gyroscope readings
154  Quaternion qGyroCur;
155  if (MathUtil::nearZero(gyro)) {
156  // if gyro reading is 0, take current quaternion to be corrected
157  qGyroCur = m_qRot;
158  } else {
159  // otherwise integrate quaternion reading
160  const Quaternion qGyroDelta{normalize(gyro), time * magn(gyro)};
161  qGyroCur = m_qRot * qGyroDelta;
162  }
163 
164  // don't bother with acceleration correction if acceleration is basically
165  // 0
166  if (MathUtil::nearZero(accel)) {
167  m_qRot = qGyroCur;
168  return;
169  }
170 
171  // gravity vector rotation
172  const Quaternion qAccelBody{0, accel};
173  const Quaternion qAccelWorld =
174  qGyroCur * qAccelBody *
175  qGyroCur.conj(); // rotates body acceleration by gyro measurements
176 
177  // correcting gyro drift with accelerometer
178  const Vector3D vecAccelWorldNorm = normalize(qAccelWorld.vec());
179  const Vector3D vecAccelGravity{0, 0, -1};
180  const Vector3D vecRotAxis =
181  cross(vecAccelWorldNorm,
182  vecAccelGravity); // rotation axis for correction rotation from
183  // estimated gravity vector (from gyro
184  // readings) to true gravity vector
185 
186 #ifdef IMUNANO33_EMBED
187  const num_t rotAngle = acosf(MathUtil::clamp(
188  dot(vecAccelGravity, vecAccelWorldNorm) /
189  (magn(vecAccelGravity) * magn(vecAccelWorldNorm)),
190  -1.0F,
191  1.0F)); // angle to rotate to correct acceleration vector
192 #else
193  const num_t rotAngle = std::acos(
194  MathUtil::clamp(dot(vecAccelGravity, vecAccelWorldNorm) /
195  (magn(vecAccelGravity) * magn(vecAccelWorldNorm)),
196  -1.0,
197  1.0)); // angle to rotate to correct acceleration vector
198 #endif
199 
200  // if angle needed to rotate is 0 or the axis to rotate around is 0, then
201  // don't bother correcting
202  if (MathUtil::nearZero(rotAngle) || MathUtil::nearZero(vecRotAxis)) {
203  m_qRot = qGyroCur;
204  return;
205  }
206 
207  // complementary filter
208  const Quaternion qAccelCur{normalize(vecRotAxis),
209  (1 - favoring) * rotAngle};
210  m_qRot = qAccelCur * qGyroCur;
211  }
